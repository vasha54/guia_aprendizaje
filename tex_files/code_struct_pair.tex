#include <bits/stdc++.h>
using namespace std;

struct Estudiante {
   string nombre;
   double estatura;
	
   Estudiante(){
      nombre="Andres Valido";
      estatura = 1.75;

      nombre=_n;
      estatura = _e;

int main() {
